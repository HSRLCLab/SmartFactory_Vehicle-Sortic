15 // This file is for defining the link class for SPI with Slave Select. The
16 // default communication for Arduino is through the ICSP connector, which uses
17 // SPI without a slave select. The LinkSPI_SS allows you to use a slave select
18 // so you can share the SPI port with other devices, or use multiple Pixys.
19 //
20 // Note, the PixySPI_SS class takes an optional argument, which is the pin
21 // number of the slave select signal you wish to use. The default pin is the
22 // SS pin (used when no argument is used.) So, for example, if you wished to
23 // use pin 14 for slave select, declare like this:
24 //
25 // PixySPI_SS pixy(14);
26 //

56  uint16_t getWord()
57  {
58  // ordering is different because Pixy is sending 16 bits through SPI
59  // instead of 2 bytes in a 16-bit word as with I2C
60  uint16_t w;
61  uint8_t c, cout = 0;
62 
63  // assert slave select
64  digitalWrite(ssPin, LOW);
65 
66  if (outLen)
67  {
68  w = SPI.transfer(PIXY_SYNC_BYTE_DATA);
69  cout = outBuf[outIndex++];
70  if (outIndex==outLen)
71  outLen = 0;
72  }
73  else
74  w = SPI.transfer(0);
75 
76  w <<= 8;
77  c = SPI.transfer(cout);
78  w |= c;
79 
80  // negate slave select
81  digitalWrite(ssPin, HIGH);
82  return w;
83  }
84 
85  uint8_t getByte() // this shouldn't be called normally
86  // It should only be called if we get out of sync, but with slave select
87  // we should stay in sync
88  {
89  uint8_t c;
90  // assert slave select
91  digitalWrite(ssPin, LOW);
92  c = SPI.transfer(0x00);
93  // negate slave select
94  digitalWrite(ssPin, HIGH);
95 
96  return c;
97  }

99  int8_t send(uint8_t *data, uint8_t len)
100  {
101  if (len>PIXY_OUTBUF_SIZE || outLen!=0)
102  return -1;
103  memcpy(outBuf, data, len);
104  outLen = len;
105  outIndex = 0;
106  return len;
107  }
108 
109  void setArg(uint16_t arg)
110  {
111  if (arg==PIXY_DEFAULT_ARGVAL)
112  ssPin = SS; // default slave select pin
113  else
114  ssPin = arg;
115  }
116 
117  private:
118  uint8_t outBuf[PIXY_OUTBUF_SIZE];
119  uint8_t outLen;
120  uint8_t outIndex;
121  uint16_t ssPin;
122 };
